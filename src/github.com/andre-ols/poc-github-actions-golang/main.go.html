    36  	ctx := gousb.NewContext()

    39  	// Debugging can be turned on; this shows some of the inner workings of the libusb package.

    42  	// OpenDevices is used to find the devices to open.

    45  		if desc.Class.String() != "hub" {

    48  			class := usbid.Classes[desc.Class]
    49  			subClass := class.SubClass[desc.SubClass]
    50  			fmt.Printf("  Class: %s\n", class)
    51  			fmt.Printf("  SubClass: %s\n", subClass)
    52  
    53  			fmt.Printf("  Serial: %s\n", desc.Device)
    54  
    55  			fmt.Printf("\n\n")
    56  
    57  			// The usbid package can be used to print out human readable information.

    61  			// to free up the memory in libusb.

    63  			// 	// This loop just uses more of the built-in and usbid pretty printing to list

    83  		// After inspecting the descriptor, return true or false depending on whether

    89  	// All Devices returned from OpenDevices must be closed.
    90  	defer func() {
    91  		for _, d := range devs {
    92  			d.Close()
    93  		}
    94  	}()
    95  
    96  	// OpenDevices can occasionally fail, so be sure to check its return value.
    97  	if err != nil {
    98  		log.Fatalf("list: %s", err)
    99  	}
   100  
   101  	for _, dev := range devs {
   102  		// Once the device has been selected from OpenDevices, it is opened
